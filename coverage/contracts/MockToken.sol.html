pragma solidity ^0.8.20;
 
import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
 
/**
 * @title MockToken
 * @dev Token mock para testes do contrato proxy
 * 
 * CARACTERÍSTICAS:
 * - Token ERC-20 simples para simular o token original
 * - Função de mint para testes
 * - Controle de acesso via Ownable
 * 
 * USO:
 * - Deployado na testnet para simular o token USDT.z original
 * - Usado nos testes unitários do contrato proxy
 * - Permite testes sem dependências externas
 */
contract MockToken is ERC20, Ownable {
    /**
     * @dev Construtor do token mock
     * @notice Mint inicial de 1.000.000 tokens para o deployer
     */
    constructor() ERC20("Mock Token", "MOCK") {
        _mint(msg.sender, 1000000 * 10 ** decimals());
    }
 
    /**
     * @dev Função para mint de tokens adicionais
     * @param to Endereço que receberá os tokens
     * @param amount Quantidade de tokens a mintar
     * @notice Apenas owner pode executar
     */
    function mint(address to, uint256 amount) public EonlyOwner {
        _mint(to, amount);
    }
}
 
/**
 * @title MockPriceFeed
 * @dev Mock do Chainlink AggregatorV3Interface para testes
 * 
 * CARACTERÍSTICAS:
 * - Simula oráculo de preço Chainlink
 * - Preço configurável para testes
 * - Interface compatível com AggregatorV3Interface
 * 
 * USO:
 * - Usado nos testes para simular preços de BNB/USD
 * - Permite testes controlados de funcionalidades de preço
 * - Evita dependências de oráculos reais nos testes
 */
contract MockPriceFeed {
    int256 private price;
    uint8 private _decimals;
 
    /**
     * @dev Construtor do mock price feed
     * @param _initialPrice Preço inicial (em wei)
     * @param decimals_ Número de casas decimais
     */
    constructor(int256 _initialPrice, uint8 decimals_) {
        price = _initialPrice;
        _decimals = decimals_;
    }
 
    /**
     * @dev Atualiza o preço para testes
     * @param _price Novo preço
     */
    function setPrice(int256 _price) external {
        price = _price;
    }
 
    /**
     * @dev Retorna o número de casas decimais
     */
    function decimals() external view returns (uint8) {
        return _decimals;
    }
 
    /**
     * @dev Retorna a descrição do price feed
     */
    function description() external pure returns (string memory) {
        return "Mock Price Feed";
    }
 
    /**
     * @dev Retorna a versão do price feed
     */
    function version() external pure returns (uint256) {
        return 1;
    }
 
    /**
     * @dev Retorna dados de uma rodada específica
     * @param _roundId ID da rodada
     */
    function getRoundData(uint80 _roundId) external view returns (
        uint80 roundId,
        int256 answer,
        uint256 startedAt,
        uint256 updatedAt,
        uint80 answeredInRound
    ) {
        return (_roundId, price, block.timestamp, block.timestamp, _roundId);
    }
 
    /**
     * @dev Retorna os dados da rodada mais recente
     */
    function latestRoundData() external view returns (
        uint80 roundId,
        int256 answer,
        uint256 startedAt,
        uint256 updatedAt,
        uint80 answeredInRound
    ) {
        return (0, price, block.timestamp, block.timestamp, 0);
    }
}
